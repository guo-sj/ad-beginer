  private async initAVSession(): Promise<void> {
    try {
      this.avSessionObj = await avSession.createAVSession(this.context, 'adVideoSession', 'video');
      await this.avSessionObj.activate();

      const metadata: avSession.AVMetadata = {
        assetId: 'ad_video_001',
        title: '广告视频',
        artist: 'AdRenderPage'
      };
      await this.avSessionObj.setAVMetadata(metadata);

      const playbackState: avSession.AVPlaybackState = {
        state: avSession.PlaybackState.PLAYBACK_STATE_IDLE,
        position: { elapsedTime: 0, updateTime: Date.now() }
      };
      await this.avSessionObj.setAVPlaybackState(playbackState);

      // 获取 Controller，监听广告服务的播放控制命令
      this.avSessionObj.getController().then((controller: avSession.AVSessionController) => {
        this.avSessionController = controller;
        this.setupAdServiceListeners();
      }).catch((err: BusinessError) => {
        HiAdLog.e(TAG, `getController failed: ${err.message}`);
      });
    } catch (err) {
      HiAdLog.e(TAG, `initAVSession failed: ${(err as BusinessError).message}`);
    }
  }

  private setupAdServiceListeners(): void {
    this.avSessionObj?.on('play', async () => { await this.playAdVideo(); });
    this.avSessionObj?.on('pause', async () => { await this.pauseAdVideo(); });
    this.avSessionObj?.on('stop', async () => { await this.stopAdVideo(); });
  }

  private async initAvPlayer(videoUrl: string): Promise<void> {
    try {
      this.avPlayer = await media.createAVPlayer();

      this.avPlayer.on('stateChange', async (state: string) => {
        if (state === 'initialized') {
          this.avPlayer!.surfaceId = this.surfaceId;
          await this.avPlayer?.prepare();
        } else if (state === 'prepared') {
          await this.avPlayer?.play();
        } else if (state === 'playing') {
          this.isVideoPlaying = true;
          this.updateAVSessionState(avSession.PlaybackState.PLAYBACK_STATE_PLAY);
        } else if (state === 'paused') {
          this.isVideoPlaying = false;
          this.updateAVSessionState(avSession.PlaybackState.PLAYBACK_STATE_PAUSE);
        }
      });

      this.avPlayer.on('error', (err) => {
        HiAdLog.e(TAG, `AVPlayer error: ${err.message}`);
      });

      this.avPlayer.on('durationUpdate', (time: number) => {
        this.videoDuration = time;
      });

      this.avPlayer.on('timeUpdate', (time: number) => {
        this.videoCurrentPos = time;
        this.updateAVSessionPosition(time);
      });

      this.avPlayer.url = videoUrl;
    } catch (err) {
      HiAdLog.e(TAG, `initAvPlayer failed: ${(err as BusinessError).message}`);
    }
  }

  private async onXComponentInit(surfaceId: string, videoUrl: string): Promise<void> {
    this.surfaceId = surfaceId;
    await this.initAvPlayer(videoUrl);
    await this.initAVSession();
  }

  private updateAVSessionState(state: avSession.PlaybackState): void {
    this.avSessionObj?.setAVPlaybackState({
      state: state,
      position: { elapsedTime: this.videoCurrentPos, updateTime: Date.now() }
    });
  }

  private updateAVSessionPosition(time: number): void {
    this.avSessionObj?.setAVPlaybackState({
      state: this.isVideoPlaying
        ? avSession.PlaybackState.PLAYBACK_STATE_PLAY
        : avSession.PlaybackState.PLAYBACK_STATE_PAUSE,
      position: { elapsedTime: time, updateTime: Date.now() }
    });
  }

  async playAdVideo(): Promise<void> {
    try { await this.avPlayer?.play(); } catch (err) {
      HiAdLog.e(TAG, `play failed: ${(err as BusinessError).message}`);
    }
  }

  async pauseAdVideo(): Promise<void> {
    try {
      if (!this.avSessionObj) return;
      const id = this.avSessionObj.sessionId;
      const ctrl = await avSession.createController(id);
      await ctrl.sendControlCommand({ command: 'pause' } as AVSessionManager.AVControlCommand);
    } catch (err) {
      HiAdLog.e(TAG, `pause failed: ${(err as BusinessError).message}`);
    }
  }

  async stopAdVideo(): Promise<void> {
    try { await this.avPlayer?.stop(); } catch (err) {
      HiAdLog.e(TAG, `stop failed: ${(err as BusinessError).message}`);
    }
  }

  private async releaseVideoResources(): Promise<void> {
    if (this.avPlayer) {
      await this.avPlayer.release();
      this.avPlayer = null;
    }
    if (this.avSessionObj) {
      await this.avSessionObj.destroy();
      this.avSessionObj = null;
    }
    this.avSessionController = null;
  }

❯ 问题1我之前该过，把ads改为@state，但是这样的话请求广告按钮一点击代码就挂了

⏺ 这个崩溃原因很典型：advertising.Advertisement 是广告 SDK 的原生对象，ArkTS 的 @State 会对数组做深层代理（deep observation），尝试对 SDK 原生对象做 Proxy 时就会崩溃。

  所以 ads 不能加 @State，应该保持 private。

  显示问题的根本原因是 Builder 函数的 this 上下文丢失，而不是 ads 缺少 @State。

  adLoaded 已经是 @State，它变为 true 时会触发重渲染，此时 this.ads 已经有数据了。真正导致父子组件不显示的是这里：

  // adComponentLayout 中：
  adRenderer: this.parentAdRenderer    // ❌ 传进去后 this 丢失，this.ads 为空

  // parentAdRenderer 中：
  adRenderer: this.childAdRenderer     // ❌ 同理

  只需要用箭头函数包一层：

  // adComponentLayout 中：
  adRenderer: () => { this.parentAdRenderer() }   // ✅

  // parentAdRenderer 中：
  adRenderer: () => { this.childAdRenderer() }    // ✅

  这样 ads 保持 private 不崩溃，且 Builder 内的 this.ads 也能正确拿到数据。
